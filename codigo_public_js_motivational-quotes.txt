    init() {
        // Show a quote immediately so the user sees it working
        this.isShowingQuote = true;
        this.updateDisplay();

        // Start alternating logic
        // We'll alternate every 45 seconds
        // Order: Quote (current) -> Date (45s) -> Quote (45s) -> Repeat
        this.startRotation();

    startRotation() {
        if (this.rotationInterval) clearInterval(this.rotationInterval);

        this.rotationInterval = setInterval(() => {
            this.isShowingQuote = !this.isShowingQuote;
            this.updateDisplay();
        }, 45000); // 45 seconds
    },

    updateDisplay() {
        const container = document.getElementById('date-quote-container');
        if (!container) return;

        container.style.opacity = '0';
        container.style.transition = 'opacity 0.5s ease-in-out';

        setTimeout(() => {
            if (this.isShowingQuote) {
                const quote = this.getRandomQuote();

                // Logic for long text marquee effect
                const isLongText = quote.text.length > 60;
                let textHtml = '';

                if (isLongText) {
                    // Marquee structure for scrolling text
                    textHtml = `
